stopIterationCallback()

Break iteration callback function.

Example
jsgui.each([1, 2, 3, 4, 5], function(index, element, stop){
   if (element > 3) stop();
});

jsgui.eac([1, 2, 3, 4, 5], function(element, index, stop){
   if (element > 3) stop();
});
